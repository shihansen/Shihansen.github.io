using namespace std ;
int n ;
double ans ;
int main()
{
	cin >> n ;
	cout << fixed << setprecision(8) ; 
	if(n <= 1e7) 
	{
		for(int i = 1 ; i <= n ; i ++) ans += (1.0 / i) ;

	}
	else cout << log(n + 1) + Euler ;
	return 0 ;
}
